		swordsmith = {
			limit = {
				prisoner = no
				OR = {
					character = FROMFROM
					AND = {
						NOT = { character = FROMFROM }
						at_location = FROMFROM
						NOR = {
							war_with = FROMFROM

								war_with = FROMFROM

				limit = { character = PREV }
